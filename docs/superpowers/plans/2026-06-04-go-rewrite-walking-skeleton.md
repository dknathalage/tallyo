# Tallyo Go Rewrite — Walking Skeleton Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up a runnable Wails v2 desktop app with one full vertical slice (Business Profile / Settings) flowing native window → Svelte-TS frontend → bound Go service → repository → audit → sqlc → modernc SQLite (goose-migrated), proving the entire rewrite architecture before any domain is ported.

**Architecture:** Standard Wails v2 pattern — embedded Svelte-TS frontend calls auto-generated Go bindings (no HTTP server, no router). Go backend: modernc.org/sqlite (pure-Go driver), goose for embedded migrations run on startup, sqlc for typed query code, a repository layer that wraps sqlc gen and writes audit rows. Clean-break data model (fresh schema, no drizzle interop).

**Tech Stack:** Go 1.26, Wails v2, Svelte 5 + TypeScript + Vite (Wails template), Tailwind CSS 4, modernc.org/sqlite, sqlc, goose, Go stdlib `testing`.

**Spec:** `docs/superpowers/specs/2026-06-04-go-rewrite-design.md`

**Scope note:** This is the first of several plans. It delivers ONLY the skeleton + Business Profile slice. Subsequent plans port clients/payers/catalog, invoices/estimates/payments, recurring, PDF, import/export, dashboard/reports, packaging. Each is its own plan.

package db

import (
	"database/sql"
	"fmt"
	"os"
	"path/filepath"

	_ "modernc.org/sqlite"
)

// DataDir resolves the application data directory.
// DATA_DIR env overrides; otherwise os.UserConfigDir()/Tallyo (matches the
// current macOS ~/Library/Application Support/Tallyo location).
func DataDir() (string, error) {
	if override := os.Getenv("DATA_DIR"); override != "" {
		return override, nil
	}
	base, err := os.UserConfigDir()
	if err != nil {
		return "", fmt.Errorf("resolve user config dir: %w", err)
	}
	return filepath.Join(base, "Tallyo"), nil
}

// Open opens a modernc SQLite connection at path and applies pragmas.
func Open(path string) (*sql.DB, error) {
	if path == "" {
		return nil, fmt.Errorf("Open: empty path")
	}
	if err := os.MkdirAll(filepath.Dir(path), 0o755); err != nil {
		return nil, fmt.Errorf("create data dir: %w", err)
	}
	conn, err := sql.Open("sqlite", path)
	if err != nil {
		return nil, fmt.Errorf("open sqlite: %w", err)
	}
	// modernc uses a single underlying handle; serialize to avoid lock churn.
	conn.SetMaxOpenConns(1)
	for _, pragma := range []string{
		"PRAGMA journal_mode = WAL",
		"PRAGMA foreign_keys = ON",
		"PRAGMA busy_timeout = 5000",
	} {
		if _, err := conn.Exec(pragma); err != nil {
			conn.Close()
			return nil, fmt.Errorf("apply %q: %w", pragma, err)
		}
	}
	return conn, nil
}
```

package db

import (
	"database/sql"
	"embed"
	"fmt"

	"github.com/pressly/goose/v3"
)

//go:embed migrations/*.sql
var migrationsFS embed.FS

// Migrate applies all embedded goose migrations to conn. Idempotent.
func Migrate(conn *sql.DB) error {
	if conn == nil {
		return fmt.Errorf("Migrate: nil conn")
	}
	goose.SetBaseFS(migrationsFS)
	if err := goose.SetDialect("sqlite3"); err != nil {
		return fmt.Errorf("set goose dialect: %w", err)
	}
	if err := goose.Up(conn, "migrations"); err != nil {
		return fmt.Errorf("goose up: %w", err)
	}
	return nil
}
```

`internal/audit/audit.go`. Use a `DBTX`-style interface so it works with `*sql.DB` AND `*sql.Tx`. The `uuid` and `created_at` are generated here (matching the current `$defaultFn` behavior). Use `github.com/google/uuid` (run `go get` for it) and `time.Now().UTC().Format(time.RFC3339)`.
```go
package audit

import (
	"context"
	"database/sql"
	"fmt"
	"time"

	"github.com/google/uuid"
)

// Execer is satisfied by *sql.DB and *sql.Tx.
type Execer interface {
	ExecContext(ctx context.Context, query string, args ...any) (sql.Result, error)
}

type Entry struct {
	EntityType string
	EntityID   int64
	Action     string
	Changes    string // JSON; defaults to "{}"
	Context    string
	BatchID    string
}

// Log writes one audit row. Every DB mutation must call this.
func Log(ctx context.Context, db Execer, e Entry) error {
	if e.EntityType == "" {
		return fmt.Errorf("audit: empty entity_type")
	}
	if e.Action == "" {
		return fmt.Errorf("audit: empty action")
	}
	changes := e.Changes
	if changes == "" {
		changes = "{}"
	}
	var batch any
	if e.BatchID != "" {
		batch = e.BatchID
	}
	_, err := db.ExecContext(ctx,
		`INSERT INTO audit_log (uuid, entity_type, entity_id, action, changes, context, batch_id, created_at)
		 VALUES (?, ?, ?, ?, ?, ?, ?, ?)`,
		uuid.NewString(), e.EntityType, e.EntityID, e.Action, changes, e.Context, batch,
		time.Now().UTC().Format(time.RFC3339),
	)
	if err != nil {
		return fmt.Errorf("audit insert: %w", err)
	}
	return nil
}
```

import (
	"context"
	"database/sql"
	"fmt"
	"time"

	"github.com/google/uuid"

	"github.com/dknathalage/tallyo/internal/audit"
	"github.com/dknathalage/tallyo/internal/db/gen"
)

type BusinessProfile struct {
	Name            string `json:"name"`
	Email           string `json:"email"`
	Phone           string `json:"phone"`
	Address         string `json:"address"`
	Logo            string `json:"logo"`
	Metadata        string `json:"metadata"`
	DefaultCurrency string `json:"default_currency"`
}

type BusinessProfileInput struct {
	Name            string `json:"name"`
	Email           string `json:"email"`
	Phone           string `json:"phone"`
	Address         string `json:"address"`
	Logo            string `json:"logo"`
	Metadata        string `json:"metadata"`
	DefaultCurrency string `json:"default_currency"`
}

type BusinessProfileRepo struct {
	db *sql.DB
}

func NewBusinessProfile(db *sql.DB) *BusinessProfileRepo {
	if db == nil {
		panic("NewBusinessProfile: nil db")
	}
	return &BusinessProfileRepo{db: db}
}

func (r *BusinessProfileRepo) Get(ctx context.Context) (*BusinessProfile, error) {
	row, err := gen.New(r.db).GetBusinessProfile(ctx)
	if err == sql.ErrNoRows {
		return nil, nil
	}
	if err != nil {
		return nil, fmt.Errorf("get business profile: %w", err)
	}
	return toDomain(row), nil
}

func (r *BusinessProfileRepo) Save(ctx context.Context, in BusinessProfileInput) error {
	if in.Name == "" {
		return fmt.Errorf("save business profile: name required")
	}

	}
